1) 테이블 생성
create table mysql.table_9th (
    seq        int(10) primary key ,
    cont       text,
    name       varchar(15),
    tel_num    int(11),
    input_date datetime
)
테이블 9th로부터 복제해서 하나 더 만들기
create table mysql.table_10th as select * from mysql.table_9th where 1=2
